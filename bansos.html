Ambil hanya angka dari input let cleanInput = input.replace(/\D/g, ''); // 2. Skema: Gunakan angka SEBELUM digit terakhir (Hapus 1 digit paling belakang) // Contoh: 3175...191 -> 3175...19 let searchTarget = cleanInput.slice(0, -1); if (searchTarget === "") { display.innerHTML = "

Input tidak valid.

"; return; } // 3. Cari di database const hasil = dbBansos.filter(item => { // Konversi NIK di Excel ke String, bersihkan dari karakter non-angka (seperti spasi/titik) // Dan pastikan menangani format scientific excel jika ada let nikDiExcel = item.NIK ? String(item.NIK).replace(/\D/g, '') : ""; // Bandingkan secara presisi return nikDiExcel === searchTarget; }); if (hasil.length > 0) { hasil.forEach(data => { // Masking NIK untuk keamanan (Menampilkan 6 digit depan dan 4 digit belakang) let rawNIK = String(data.NIK); const maskedNIK = rawNIK.length > 10 ? rawNIK.replace(/(\d{6})\d+(\d{4})/, "$1******$2") : "Terdaftar"; display.innerHTML += `
${data.Nama || 'Tanpa Nama'}

${data.Jenis || 'Bantuan'} RT: ${data.RT || '-'}
`; }); } else { display.innerHTML = `
